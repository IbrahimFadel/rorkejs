import {Text as PixiText, TextStyle} from 'pixi.js';

/**
 * Rorke Text
 * @class
 */
export default class Text {
	/**
	 *
	 * @param {number} x
	 * @param {number} y
	 * @param {string} text text value
	 * @param {object} options styling(font, color, etc. )
	 * @param {object} rorke reference values from game instance
	 */
	constructor(x, y, text, options, rorke) {
		this.pixi = {
			text: undefined,
		};
		this.rorke = {
			app: rorke.pixi.app,
		};
		this.x = x;
		this.y = y;
		this.text = text;
		this.options = options;
		this.angle = 0;
	}

	/**
	 * Create a pixi text object with/without styles. Add it to the canvas
	 * @return {object} Rorke Text
	 */
	add() {
		let style;
		let newText;
		if (this.options) {
			style = new TextStyle(this.options);
			newText = new PixiText(this.text, style);
		} else {
			newText = new PixiText(this.text);
		}
		newText.x = this.x;
		newText.y = this.y;
		this.pixi.text = newText;
		this.rorke.app.stage.addChild(newText);
		return this;
	}

	/**
	 * Update the pixi text x and y values to the x and y value of the rorke text
	 */
	updatePos() {
		this.pixi.text.x = this.x;
		this.pixi.text.y = this.y;
	}

	/**
	 * Update the pixi text angle value to the angle of the rorke text
	 */
	updateRot() {
		this.pixi.text.angle = this.angle;
	}

	/**
	 * Update all pixi text values
	 */
	update() {
		this.updatePos();
		this.updateRot();
	}
}
